			ui.Badge(ui.BadgeProps{Variant: ui.BadgeSecondary}, ui.Icon("circle-check", "size-3 text-emerald-500"), "Saved"),

		g.Div(g.Class("flex items-center gap-3"),
			ui.Skeleton("size-10 rounded-full"),
			g.Div(g.Class("flex flex-col gap-2"), ui.Skeleton("h-3 w-32"), ui.Skeleton("h-3 w-24")),
		),

}
